expr:
    - KEYWORD:varcreate IDENTIFIER (COMMA IDENTIFIER)* EQ expr
    - compexpr ((KEYWORD:logicaland|KEYWORD:logicalor) compexpr)*

compexpr:
    - KEYWORD:logicalnot compexpr
    - arithexpr ((EQEQUALS|BANGEQUALS|LESSTHAN|LTEQUALS|GREATERTHAN|GTEQUALS) arithexpr)*

arithexpr:
    - term ((PLUS|DASH) term)*

term:
    - raised ((ASTRISK|FSLASH) raised)*

raised:
    - factor (CARAT factor)*

factor:
    - (PLUS|DASH) factor
    - call

call:
    - IDENTIFIER LPAREN (expr (COMMA expr)*)? RPAREN
    - atom

atom:
    - INT|FLOAT|STRING
    - LPAREN expr RPAREN
    - IDENTIFIER (COMMA IDENTIFIER)* EQUALS expr
    - IDENTIFIER
    - funcdef
    - handler

funcdef:
    - KEYWORD:createfunc LPAREN (IDENTIFIER (COMMA IDENTIFIER)*)? RPAREN codeblock

handler:
    - KEYWORD:handler codeblock

codeblock:
    - LCURLY (EOL* (expr (EOL expr))? EOL*) RCURLY
