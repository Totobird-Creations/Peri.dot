Literal  : INT FLOAT STRING
Operator : PLUS MINUS TIMES DIVBY POW SQRT


notraceexpr:
  - KEYWORD:var IDENTIFIER EQUALS expr
  - expr

expr:
  - arithexpr

arithexpr:
  - factor (factor (PLUS|MINUS|TIMES|DIVBY|POW|SQRT))*
  - factor (factor (EQUALEQ|NOTEQ|LSSTHN|LSSTHNEQ|GRTTHN|GRTTHNEQ))

factor:
  - (PLUS|MINUS) factor
  - LPAREN expr RPAREN
  - atom

atom:
  - INT
  - FLOAT
  - STRING
  - IDENTIFIER
