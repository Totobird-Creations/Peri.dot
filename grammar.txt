expr:
    - KEYWORD:varcreate IDENTIFIER (COMMA IDENTIFIER)* EQ expr
    - comp-expr ((KEYWORD:logicaland|KEYWORD:logicalor) comp-expr)*

comp-expr:
    - KEYWORD:logicalnot comp-expr
    - arith-expr ((EQEQUALS|BANGEQUALS|LESSTHAN|LTEQUALS|GREATERTHAN|GTEQUALS) arith-expr)*

arith-expr:
    - term ((PLUS|DASH) term)*

term:
    - raised ((ASTRISK|FSLASH) raised)*

raised:
    - factor (CARAT factor)*

factor:
    - (PLUS|DASH) factor
    - call

call:
    - IDENTIFIER LPAREN (expr (COMMA expr)*)? RPAREN
    - atom

atom:
    - INT|FLOAT|STRING
    - LPAREN expr RPAREN
    - IDENTIFIER (COMMA IDENTIFIER)* EQUALS expr
    - IDENTIFIER
